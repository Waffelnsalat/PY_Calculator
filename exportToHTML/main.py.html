#define calculation
def button_calc():
    value = str(counter["text"])
    try:
        result = eval(value)
        counter["text"] = str(result)
    except:
        print("SyntaxError")
        counter["text"] = "Syntax Error"

#define delete button
def button_del():
    value = str(counter["text"])
    try:
        counter["text"] = value.rstrip(value[-1])
    except:
        print("Nothing to delete")
