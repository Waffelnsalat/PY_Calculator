    if "+" in value:
        eco1 = value.split("+")
        result = int(eco1[0])
        count = len(eco1)
        for x in range(count - 1):
            result = result + int(eco1[x + 1])
    elif "-" in value:
        eco1 = value.split("-")
        result = int(eco1[0])
        count = len(eco1)
        for x in range(count - 1):
            result = result - int(eco1[x + 1])
    counter["text"] = result
